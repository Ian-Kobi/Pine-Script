indicator(title='Money Flow', shorttitle='MF', format=format.volume, precision=0, timeframe='')

src = input(close, title='Source')
MF = ta.cum(src*(math.sign(ta.change(src))*volume))

plot(MF, color=color.new(color.yellow, 0))
